# Ukazka konfiguracniho souboru
# Prepiste si cesty k souborum dir=... a ulozte tento soubor
# do vaseho home pod nazvem .ccreg_client.conf
# Konfiguracni soubor lze take vytvorit z klienta prikazem 'config create'.
#

[connect]
dir=/home/username/certificates
host = epp-test.ccreg.nic.cz
port = 700
ssl_cert = %(dir)s/epp_test_ccreg_nic_cz_cert.pem
ssl_key  = %(dir)s/epp_test_ccreg_nic_cz_key.pem
username = MYUSERNAME
password = MYPASSWORD
# socket = IPv6     # paltne hodnoty jsou IPv4 a IPv6

[session]
poll_ack = off
confirm_send_commands = on
validate = on
schema = /home/username/epp_client/ccReg/schemas/all-1.0.xsd
colors = yes
verbose = 2
